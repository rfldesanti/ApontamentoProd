Sub MUC2_PROC()

If mq.Value = "" Then
    mq.ClearContents

    ApontaOP.Unprotect "TESTE"

    mq.Validation _
        .Modify xlValidateList, xlValidAlertStop, _
        xlBetween, "=mq_db!$A$2:$A$6"
  
    ApontaOP.Protect "TESTE"
Else
    If mq.Locked = False And IsError(Application.Match(mq.Value, mq_db.Range("A:A"), 0)) Then
        mq.ClearContents

        ApontaOP.Unprotect "TESTE"

        mq.Validation _
            .Modify xlValidateList, xlValidAlertStop, _
            xlBetween, "=mq_db!$A$2:$A$6"
  
        ApontaOP.Protect "TESTE"
    End If
End If

End Sub
