Sub RegistroNovo()

If Not IsError(Application.Match(ordem_prod, db.Range("A:A"), 0)) And _
Not IsError(Application.Match(id_funcionario, db_funcionarios.Range("A:A"), 0)) Then
    Call ContaLinhas
    registros.Cells(UltimaLinha, 1).Value = ordem_prod
Else
    MsgBox "Ordem e\ou funcionário não encontrados - Informe novos valores"
    ordem_prod.ClearContents
    Call botao
End If

End Sub
